void CBox(char x0,char y0,char x1,char y1,char co)
{
char i,c,w;
c=abs(co/16-co%16)+1;
w=x1-x0+1;
if((co/16-co%16)<0)
  for(i=0;i<c;i++)
    {
    SetFgColor(co/16+i);
    Block(x0+w*i/c,y0,x0+w*(i+1)/c,y1,1);
    }
else
  for(i=0;i<c;i++)
    {
    SetFgColor(co/16-i);
    Block(x0+w*i/c,y0,x0+w*(i+1)/c,y1,1);
    }
}

int strval()
//用来将字符串转化成数值
{
char i;
int val;
ch=i=val=0;
for(;i<strlen(string);i++)
  {
  ch=string[i];
  val=val*10+(ch-48);
  }

}
int inputnum(char len,char y,char x)

{

    {

void drawLine(char x,char y,char sl,char el,int deg)
{
int dx,dy;
deg=deg%360;
if(deg<90)
  {
  dx=1;
  dy=-1;
  deg=90-deg;
  }
else if(deg<180)
  {
  dx=1;
  dy=1;
  deg=deg-90;
  }
else if(deg<270)
  {
  dx=-1;
  dy=1;
  deg=270-deg;
  }
else
  {
  dx=-1;
  dy=-1;
  deg=deg-270;
  }
Line(x+dx*sl*Cos(deg)/1024,y+dy*sl*Sin(deg)/1024,x+dx*el*Cos(deg)/1024,y+dy*el*Sin(deg)/1024,65);
}
void QWatch(char x,char y,char r)
{
//中心坐标,钟表半径
char i,ms,sl,ml,hl,sec;
sec=255;
//这儿设定分针,秒针,时针的长度
sl=5*r/6;
ml=4*r/5;
hl=r/2;
while(Inkey()-27)
  {
  GetTime(time);
  if(time.second-sec)
    {
    sec=time.second;
    Block(x-r,y-r,x+r,y+r,0);
    for(i=0;i<60;i++)if(i%5)drawLine(x,y,r-1,r,i*6);
    else drawLine(x,y,r-3,r,i*6);
    drawLine(x,y,0,sl,time.second*6);
    drawLine(x,y,0,ml,time.minute*6);
    drawLine(x,y,0,hl,time.hour*30+time.minute/2);
    Refresh();
    }
  }
}

char dayofweek(int y,char m,char d)
//已知日期求星期
{
if(m<3)y--;
return (y+y/4-y/100+y/400+wk[m-1]+d)%7;
}
char isInYear(int year)
//判断闰年(isIntercalaryYear)
{
if(year%4&&year%100)return 0;
else if(!(year%400))return 0;
else return 1;
}

char sCal(int yy,char mm,char dd)
//showCalendar
{

ww=yy%12;

*(stra+4)=0;
memcpy(stra,DZ+ww*4-16,4);

sprintf(stra,"%d",yy);
TextOut(6,39,stra,1);
sprintf(stra,"%d",mm);

d=dd%7;

if(!d)d=7;
x=ww-d+1;
if(x>127)x=x+7;
strb[0]=x;
d=1;
y=0;
do
  {
  sprintf(stra,"%d",d);
  TextOut(40+17*x,17+11*y,stra,1);
  if(d==dd)
    {
    sx=x;
    sy=y;
    }
  if(++x>6)
    {
    x=0;
    if(++y>4)y=0;
    }
  d++;
  }while(d<=dmax[mm-1]);
return strb[0];
}
char ctrlCal(int yy,char mm,char dd)
//controlCalendar
{

do
  {
  if(WID-64)
    {
    WID=64;
    aa=sCal(yy,mm,dd);
    Block(36+17*sx,17+11*sy,53+17*sx,28+11*sy,2);
    }
  Refresh();
  cha=CheckKey(128);
  Delay(50);
  if(cha=='q')exit(0);
  else if(cha=='a')frmAbout();
  else if(cha==25||cha==28)CallHelp(64);
  else if(cha==14||cha==19)
    {
    dmax[1]=28+isInYear(yy);
    if(cha-19)
      {
      if(mm-12)mm++;
      else
        {
        mm=1;
        if(yy-2199)yy++;
        else yy=1753;
        }
      }
    else
      {
      if(mm-1)mm--;
      else
        {
        mm=12;
        if(yy-1753)yy--;
        else yy=2199;
        }
      }
    bb=sy*7+sx-aa+1;
    while(bb>dmax[mm-1])bb--;
    aa=sCal(yy,mm,bb);
    Block(36+17*sx,17+11*sy,53+17*sx,28+11*sy,2);
    continue;
    }
  else if(cha==22||cha==23||cha==21||cha==20)
    {
    Block(36+17*sx,17+11*sy,53+17*sx,28+11*sy,2);
    do
      {
      if(cha==22)sx++;
      else if(cha==23)sx--;
      else if(cha-20)sy++;
      else sy--;
      if(sx>127)sx=6;
      else if(sx>6)sx=0;
      if(sy>127)sy=4;
      else if(sy>4)sy=0;
      if(!sy&&sx<aa)bb=35+sx-aa+1;
      else bb=sy*7+sx-aa+1;
    }while(bb>dmax[mm-1]);
    Block(36+17*sx,17+11*sy,53+17*sx,28+11*sy,2);
    }
  }while(cha-27&&cha-13);
if(cha-27)
  {
  addt.year=yy;
  addt.month=mm;
  addt.day=(sy*7+sx-aa+1);
  ClearScreen();
  ReleaseKey(128);
  Delay(50);
  return addt.day;
  }
else
  {
  ClearScreen();
  sx=0;
  ReleaseKey(128);
  Delay(100);
  }
}

  else if(ch=='s')
    {
    wz=4;
    DCircle(wz);
    }
  else if(ch==13)
    {
    inum=0;
    if(wz==2)
      {
      do
        {
        inum=InputX(wz);
        }while(inum>23&&inum-100);
      if(inum-100)addt.hour=inum;
      else 
        {
        sprintf(stra,"%d ",addt.hour);
        TextOut(39,37,stra,1);
        }
      }
    else if(wz==3)
      {
      do
        {
        inum=InputX(wz);
        }while(inum>59&&inum-100);
      if(inum-100)addt.minute=inum;
      else
        {
        sprintf(stra,"%d ",addt.minute);
        TextOut(39,50,stra,1);
        }
      }
    else
      {
      do
        {
        inum=InputX(wz);
        }while(inum>59&&inum-100);
      if(inum-100)addt.second=inum;
      else
        {
        sprintf(stra,"%d ",addt.second);
        TextOut(39,63,stra,1);
        }
      }
    Refresh();
    Delay(200);
    }
  else if(ch==20)
    {
    if(wz==2)wz=4;
    else wz--;
    DCircle(wz);
    Delay(50);
    }
  else if(ch==21)
    {
    if(wz==4)wz=2;
    else wz++;
    DCircle(wz);
    Delay(50);
    }
  else if(ch=='z')
    {
    GetTime(time);
    time.hour=addt.hour;
    time.minute=addt.minute;
    time.second=addt.second;
    SetTime(time);
    TextOut(80,65,"设置完毕！",1);
    StrTime(time);
    Block(12,24,60,37,0);
    TextOut(12,24,strt,1);
    Refresh();
    Delay(200);
    TextOut(80,65,"          ",1);
    Refresh();
    }
  }
ReleaseKey(128);
Delay(100);
}

{

else if(ch==25||ch==28)CallHelp(frmID);
else if(ch=='a')frmAbout();
else if(ch=='w')STime();
else if(ch=='e')SDate();
else if(ch=='d')SaveDate();
else if(ch=='s')SaveTime();
else if(ch=='c')LoadDate();
else if(ch=='x')LoadTime();

}
